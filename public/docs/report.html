1. Описание страниц приложения

Веб-приложение "BookStore" состоит из следующих страниц:

Страница Файл Назначение
Главная страница index.html Входная точка приложения. Содержит hero-секцию с приветствием, блоки новинок и популярных книг, а также навигацию по жанрам. Привлекает пользователя и направляет его в каталог.
Каталог catalog.html Основная страница магазина. Содержит сетку товаров (книг) с возможностью фильтрации по жанрам: фантастика, романы, детективы, приключения, учебные материалы. Включает боковую панель фильтров и фильтрацию по цене.
Корзина cart.html Страница управления корзиной. Отображает добавленные товары с возможностью изменения количества и удаления. Показывает итоговую сумму заказа и кнопку перехода к оформлению.
Оформление заказа checkout.html Страница оформления заказа. Содержит форму ввода контактных данных (ФИО, телефон, email), адреса доставки, выбора способа доставки и оплаты. Показывает сводку заказа.
Регистрация register.html Страница регистрации нового пользователя. Форма содержит поля: имя, email, пароль, подтверждение пароля. Включает валидацию данных.
Вход login.html Страница авторизации пользователя. Форма с полями email и пароль для входа в личный кабинет.

Каждая страница содержит общие элементы:

  • Шапка (header) — логотип, ссылка на корзину с счётчиком товаров, ссылка на вход/регистрацию
  • Навигация (nav) — меню с ссылками на главную, каталог и жанры книг
  • Основной контент (main) — уникальное содержимое страницы
  • Подвал (footer) — информация о магазине, ссылки на разделы, контакты

5. JS код взаимодействия с серверной частью (AJAX)

5.1. Класс ApiService

Сервис для выполнения HTTP-запросов к серверу через AJAX (XMLHttpRequest).

/**
 * Сервис для работы с API сервера
 * Все запросы выполняются через AJAX (XMLHttpRequest)
 */
class ApiService {
    constructor(baseUrl = '/api') {
        this.baseUrl = baseUrl;
    }

    /**
     * Базовый метод для выполнения AJAX запросов
     * @param {string} method - HTTP метод (GET, POST, PUT, DELETE)
     * @param {string} endpoint - Конечная точка API
     * @param {object} data - Данные для отправки
     * @returns {Promise} - Промис с результатом
     */
    request(method, endpoint, data = null) {
        return new Promise((resolve, reject) => {
            // Создание объекта XMLHttpRequest
            const xhr = new XMLHttpRequest();
            
            // Настройка запроса
            xhr.open(method, this.baseUrl + endpoint, true);
            xhr.setRequestHeader('Content-Type', 'application/json');
            xhr.setRequestHeader('Accept', 'application/json');
            
            // Добавление токена авторизации если есть
            const token = StorageService.get('authToken');
            if (token) {
                xhr.setRequestHeader('Authorization', 'Bearer ' + token);
            }
            
            // Обработчик успешного ответа
            xhr.onload = function() {
                if (xhr.status >= 200 && xhr.status < 300) {
                    try {
                        const response = JSON.parse(xhr.responseText);
                        resolve(response);
                    } catch (e) {
                        resolve(xhr.responseText);
                    }
                } else {
                    reject({
                        status: xhr.status,
                        message: xhr.statusText,
                        response: xhr.responseText
                    });
                }
            };
            
            // Обработчик ошибки сети
            xhr.onerror = function() {
                reject({
                    status: 0,
                    message: 'Ошибка сети',
                    response: null
                });
            };
            
            // Обработчик таймаута
            xhr.ontimeout = function() {
                reject({
                    status: 0,
                    message: 'Превышено время ожидания',
                    response: null
                });
            };
            
            // Установка таймаута (30 секунд)
            xhr.timeout = 30000;
            
            // Отправка запроса
            if (data) {
                xhr.send(JSON.stringify(data));
            } else {
                xhr.send();
            }
        });
    }
}

5.2. Методы для работы с API

// GET запрос - получение списка книг
getBooks(filters = {}) {
    let query = this.buildQueryString(filters);
    return this.request('GET', '/books' + query);
}

// GET запрос - получение одной книги
getBook(id) {
    return this.request('GET', '/books/' + id);
}

// POST запрос - регистрация пользователя
register(userData) {
    return this.request('POST', '/auth/register', userData);
}

// POST запрос - авторизация
login(credentials) {
    return this.request('POST', '/auth/login', credentials);
}

// POST запрос - создание заказа
createOrder(orderData) {
    return this.request('POST', '/orders', orderData);
}

// PUT запрос - обновление корзины
updateCart(cartData) {
    return this.request('PUT', '/cart', cartData);
}

// DELETE запрос - удаление товара из корзины
removeFromCart(bookId) {
    return this.request('DELETE', '/cart/items/' + bookId);
}

5.3. Примеры использования AJAX в контроллерах

// Пример загрузки книг в CatalogController
async loadBooks(filters) {
    this.showLoading(this.container);
    
    try {
        // AJAX запрос через ApiService
        const response = await this.apiService.getBooks(filters);
        
        // Преобразование данных в модели Book
        const books = response.data.map(data => Book.fromJSON(data));
        
        // Рендеринг результата
        this.renderBooks(books);
    } catch (error) {
        this.showError(this.container, 'Ошибка загрузки каталога');
    }
}

// Пример добавления в корзину
async addToCart(bookId) {
    try {
        // Получение книги с сервера
        const bookData = await this.apiService.getBook(bookId);
        const book = Book.fromJSON(bookData);
        
        // Добавление в модель корзины
        this.cart.addItem(book, 1);
        
        // Сохранение на сервере
        await this.apiService.updateCart(this.cart.toJSON());
        
        // Обновление UI
        this.updateCartCounter();
        this.showSuccess(document.body, 'Товар добавлен в корзину');
    } catch (error) {
        this.showError(document.body, 'Ошибка добавления в корзину');
    }
}

// Пример оформления заказа
async submitOrder(formData) {
    try {
        // Создание заказа из корзины
        const order = Order.createFromCart(this.cart, formData);
        
        // Валидация через модель
        const validation = order.validate();
        if (!validation.isValid) {
            throw new Error(validation.errors.join(', '));
        }
        
        // Отправка на сервер
        const response = await this.apiService.createOrder(order.toJSON());
        
        // Очистка корзины
        this.cart.clear();
        await this.apiService.updateCart(this.cart.toJSON());
        
        // Показ подтверждения
        this.showOrderConfirmation(response);
    } catch (error) {
        this.showError(this.form, error.message);
    }
}

6.4. Дополнительные адаптивные техники

  • Гибкие изображения: max-width: 100%; height: auto;
  • Относительные единицы: использование rem, em, % вместо px
  • Flexbox: основной инструмент для гибкой раскладки
  • CSS Grid: для сеток товаров с автоматическим переносом
  • Скрытие/показ элементов: классы .hidden-mobile, .hidden-desktop
  • Touch-friendly: увеличенные области касания для мобильных (min 44x44px)
